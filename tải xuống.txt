-- ===== CACHE =====
local monsterCache = {best = nil, time = 0}
local fruitCache = {best = nil, time = 0}

-- ===== HÀM TÌM QUÁI =====
local function getNearestMonster()
    if tick() - monsterCache.time < 0.3 then return monsterCache.best end
    monsterCache.time = tick()
    local best, bestDist = nil, math.huge
    for _, v in pairs(workspace:GetDescendants()) do
        if v:IsA("Model") and v:FindFirstChild("Humanoid") and v:FindFirstChild("HumanoidRootPart") then
            if v ~= character and v.Humanoid.Health > 0 and v.Name:match("NPC") or v.Name:match("Boss") or v.Name:match("Monster") then
                local d = (rootPart.Position - v.HumanoidRootPart.Position).Magnitude
                if d < bestDist and d < config.FarmRadius then
                    best = v; bestDist = d
                end
            end
        end
    end
    monsterCache.best = best
    return best
end

-- ===== HÀM TÌM TRÁI =====
local function getNearestFruit()
    if tick() - fruitCache.time < 0.3 then return fruitCache.best end
    fruitCache.time = tick()
    local best, bestDist = nil, math.huge
    for _, v in pairs(workspace:GetDescendants()) do
        if v:IsA("Part") and (v.Name:find("Fruit") or v.Name:find("Banana") or v:FindFirstChild("Handle")) then
            local name = v.Name
            if v.Parent and v.Parent:FindFirstChild("Name") then name = v.Parent.Name end
            local isPremium = false
            for _, p in pairs(config.PriorityFruits) do
                if name:find(p) then isPremium = true; break end
            end
            local d = (rootPart.Position - v.Position).Magnitude
            if isPremium and d < bestDist and d < 600 then
                best = v; bestDist = d
            elseif not isPremium and d < bestDist and d < 400 then
                best = v; bestDist = d
            end
        end
    end
    fruitCache.best = best
    return best
end

-- ===== TẤN CÔNG =====
local function attackMonster(target)
    if not target then return end
    rootPart.CFrame = CFrame.new(rootPart.Position, target.HumanoidRootPart.Position)
    local atk = replicatedStorage:FindFirstChild("Combat") or replicatedStorage:FindFirstChild("Attack")
    if atk then atk:FireServer("Click")
    else
        for _, rem in pairs(replicatedStorage:GetDescendants()) do
            if rem:IsA("RemoteEvent") and (rem.Name:find("Attack") or rem.Name:find("Hit")) then
                rem:FireServer(target); break
            end
        end
    end
end

-- ===== NHẶT TRÁI =====
local function collectFruit(fruit)
    if not fruit then return end
    rootPart.CFrame = CFrame.new(fruit.Position + Vector3.new(0, 2.5, 0))
    local touch = fruit:FindFirstChild("TouchInterest")
    if touch then touch:FireServer() end
    for _, rem in pairs(replicatedStorage:GetDescendants()) do
        if rem:IsA("RemoteEvent") and (rem.Name:find("Collect") or rem.Name:find("Pickup")) then
            rem:FireServer(fruit); break
        end
    end
    local name = fruit.Name
    if fruit.Parent and fruit.Parent:FindFirstChild("Name") then name = fruit.Parent.Name end
    player:Chat("🍌 Nhặt: " .. name)
end
